def VisualizeSDF(object, z_height, resolution=100, iso_surface=0.0):
    # Get bounding box from the node
    min_pt, max_pt = object.bounding_box()
    min_x, min_y = min_pt.x, min_pt.y
    max_x, max_y = max_pt.x, max_pt.y

    # Calculate sampling grid size based on resolution and aspect ratio
    width = max_x - min_x
    height = max_y - min_y

    if width > height:
        nx = resolution
        ny = int(resolution * (height / width)) if width > 0 else resolution
    else:
        ny = resolution
        nx = int(resolution * (width / height)) if height > 0 else resolution

    nx = max(1, nx)
    ny = max(1, ny)

    # Generate world-space sample coordinates
    xs = np.linspace(min_x, max_x, nx)
    ys = np.linspace(min_y, max_y, ny)
    xv, yv = np.meshgrid(xs, ys)

    # object.prepare(pv.Vec3(xs,ys,min(xs,ys)), 1.0)

    # Evaluate SDF at each grid point
    sdf_values = np.zeros_like(xv)
    for i in range(ny):
        for j in range(nx):
            sdf_values[i, j] = object.evaluate(xv[i, j], yv[i, j], z_height)

    # Set up the plot
    fig, ax = plt.subplots(figsize=(8, 6))

    # Use TwoSlopeNorm to center the colormap at 0 (surface boundary)
    # This helps distinguish inside (negative) vs outside (positive)
    v_min, v_max = sdf_values.min(), sdf_values.max()
    norm = None
    if v_min < 0 < v_max:
        norm = mcolors.TwoSlopeNorm(vmin=v_min, vcenter=0., vmax=v_max)

    # Plot the heat map
    im = ax.imshow(sdf_values,
                   extent=[min_x, max_x, min_y, max_y],
                   origin='lower',
                   cmap='seismic',
                   norm=norm)

    plt.colorbar(im, ax=ax, label='Signed Distance')

    # Draw the specific iso-surface contour
    ax.contour(xv, yv, sdf_values, levels=[iso_surface], colors='black', linewidths=1.5, linestyles='dashed')

    ax.set_title(f'SDF Visualization at Z={z_height}')
    ax.set_xlabel('X')
    ax.set_ylabel('Y')

    plt.show()
